# let's put the plot code inside a function so we can re-use it later
def plot_duration_log(data, msg):
    '''
    A function to plot a log scale histogram for the duration feature
    
    inputs:
        data : the dataframe that contain the duration column
        msg : the customized msg to be printed under the plot
    outputs:
        a log scale histogram for the duration column, with the msg sent, and the mean duration is printed. 
    '''
    log_binsize = 0.0125
    
    # 1.7 is the log(minimum duration from the data-set)
    bins = 10 ** np.arange(1.7, np.log10(data['duration_sec'].max())+log_binsize, log_binsize)

    plt.figure(figsize=standard)
    plt.hist(data = data, x = 'duration_sec', bins = bins)
    plt.xscale('log')
    plt.xlabel('Duration (sec) '+msg)
    plt.axvline(data['duration_sec'].mean(), color='k', linestyle='dashed', linewidth=1)
    plt.suptitle("Histogram of duration in seconds", y = 1.04)
    plt.show()
#     print(" The mean of ride duration is  = ", data['duration_sec'].mean())
plot_duration_log(data, msg = 'for all users')
